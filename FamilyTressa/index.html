Algoritma

function traversePreOrder(node) {
  console.log(node.nama);
  node.anak.forEach(child => traversePreOrder(child));
}

function findNode(node, name) {
  if(node.nama === name) return node;
  for(let child of node.anak){
    const found = findNode(child, name);
    if(found) return found;
  }
  return null;
}

function addChild(parentName, childName, avatar) {
  const parentNode = findNode(familyTree, parentName);
  if(!parentNode){
    familyTree.anak.push({nama: childName, avatar: avatar, anak: []});
  } else {
    parentNode.anak.push({nama: childName, avatar: avatar, anak: []});
  }
}

Struktur Data

Node {
  nama: string
  avatar: string (path gambar)
  anak: array of Node
}
